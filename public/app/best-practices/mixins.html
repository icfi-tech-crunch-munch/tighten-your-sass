Mixins

Mixins can quickly add bloat to your css.

@mixin clearfix {
  &:before,
  &:after {
    content: " "; // 1
    display: table; // 2
  }
  &:after {
    clear: both;
  }
}

.row {
    @include clearfix();
}

.column {
    @include clearfix();
}

compiles to:

.row {
  &:before,
  &:after {
    content: " "; // 1
    display: table; // 2
  }
  &:after {
    clear: both;
  }
}

.column {
  &:before,
  &:after {
    content: " "; // 1
    display: table; // 2
  }
  &:after {
    clear: both;
  }
}

Create a class and use @extend instead:

@mixin clearfix {
  &:before,
  &:after {
    content: " "; // 1
    display: table; // 2
  }
  &:after {
    clear: both;
  }
}

.clearfix {
    @include clearfix();
}

.row {
    @extend .clearfix;
}

.column {
    @extend .clearfix;
}

compiles to:

.clearfix,
.row,
.column {
  &:before,
  &:after {
    content: " "; // 1
    display: table; // 2
  }
  &:after {
    clear: both;
  }
}
