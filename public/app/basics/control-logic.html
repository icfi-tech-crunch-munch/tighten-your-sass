Control Logic

Sass provides a handful of control logic directives.

@if

@mixin absolutely ($top:null, $right:null, $bottom:null, $left:null) {
    position: absolute;
    @if ($top)      { top: $top; }
    @if ($right)    { right: $right; }
    @if ($bottom)   { bottom: $bottom; }
    @if ($left)     { left: $left; }
}

.thumbnail {
    @include absolutely($bottom: 0, $right: 0);
}

Compiles to:

.thumbnail {
    position: absolute;
    bottom: 0;
    right: 0;
}

@for

@for $index from 1 through 3 {
    .item-#{$index} { width: 10px * $index; }
}

Compiles to:

.item-1 {
    width: 10px;
}
.item-2 {
    width: 20px;
}
.item-3 {
    width: 30px;
}

@each

@each $icon in error, warning, info {
    .icon-#{$icon} {
        background-image: url('/images/icon-#{$icon}.png');
    }
}

Compiles to:

.icon-error {
    background-image: url('/images/icon-error.png');
}
.icon-warning {
    background-image: url('/images/icon-warning.png');
}
.icon-info {
    background-image: url('/images/icon-info.png');
}

@for

@for $index from 1 through 3 {
    .item-#{$index} { width: 10px * $index; }
}

Compiles to:

.item-1 {
    width: 10px;
}
.item-2 {
    width: 20px;
}
.item-3 {
    width: 30px;
}

@while

$cols: 1;
@while $cols <= 8 {
    .col-#{$cols} {
        width: 100% * ($cols / 8);
    }
    $cols: $cols * 2;
}

Compiles to:

.col-1 {
  width: 12.5%; }

.col-2 {
  width: 25%; }

.col-4 {
  width: 50%; }

.col-8 {
  width: 100%; }
